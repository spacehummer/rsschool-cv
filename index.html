/**
 * Extract all strings of Array as text node wrapped in specified child elements in node.
 * @param nodeToUpdate        {Object}  - the node to which child elements items will be added.
 * @param elementsArray       {Array}   - Array of strings, which will be converted into text nodes.
 * @param elementName         {String}  - name of elements (tag), which will be used for wrap
 *                                        every text node.
 * @param elementsClassesList {Array}   - name of classes for each element, in with text nodes
 *                                        will be wrapped.
 * @description If elementsClassesList specified, add classes from list of classes
 *              for each child element.
 */
function addAllFieldsAsTextNode(
  nodeToUpdate,
  elementsArray,
  elementName,
  elementsClassesList,
) {
  for (let i = 0; i < elementsArray.length; i++) {
    nodeToUpdate.appendChild(document.createElement(elementName));
  }
  /* Append elements and work with classes only if node have children.
  * TODO: error handling!
  * */
  if (!nodeToUpdate.hasChildNodes()) {
    return;
  }
  const children = nodeToUpdate.childNodes;
  for (let i = 0; i < children.length; ++i) {
    children[i].appendChild(document.createTextNode(elementsArray[i]));
    if (elementsClassesList[i] !== '') {
      children[i].classList.add(elementsClassesList[i]);
    }
  }
}
